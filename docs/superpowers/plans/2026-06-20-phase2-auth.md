# Peakstats Phase 2 — Auth Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Let an athlete sign in with Strava, persist their tokens server-side, hold a session, read their profile, and disconnect — proving the full auth loop end to end.

**Architecture:** FastAPI owns auth. `GET /auth/strava/login` redirects to Strava; `GET /auth/strava/callback` exchanges the code for tokens, upserts the athlete + tokens into Supabase (service-role key, bypasses RLS), and sets a **signed HttpOnly cookie** (`ps_session`, signed with `SESSION_SECRET` via itsdangerous). All authenticated requests carry that cookie; the backend resolves it to an `athlete_id` and scopes every query. The SPA never sees the Strava client secret or any token. Strava tokens stay server-only; RLS is a deny-client backstop (the client never receives a Supabase JWT). The SPA redirects to the backend for login, then fetches `/athlete` with `credentials: "include"`, and shows a minimal authenticated `/app` home with Logout / Disconnect.

**Tech Stack:** Python 3.12 + FastAPI + httpx + itsdangerous + pytest (backend); React 19 + Vite + TypeScript + Vitest + Testing Library (frontend); Supabase Postgres via PostgREST; Strava OAuth.

## Global Constraints

- Python 3.12; Node 20+ (dev machine has 25).
- Strava client id/secret + Supabase service-role key live **only** on the backend (Render). Frontend gets only the API base URL (+ Supabase anon key, unused for data in this phase).
- Data stored **metric**; no conversion server-side (not exercised in this phase).
- All tables have RLS enabled. The backend uses the **service-role key** (bypasses RLS) for every read/write; the client never receives a Supabase JWT, so client-side direct reads stay denied.
- Session = backend-issued **signed HttpOnly cookie** `ps_session`. Cross-site in prod, so `SameSite=None; Secure`; same-site in local dev, so `SameSite=Lax; Secure=false`. CORS already runs `allow_credentials=True`; the SPA fetches with `credentials: "include"`.
- Strava scopes: `read,activity:read_all`. **One** Strava app (Strava allows a single app per account) with callback domain `peakstats-api.onrender.com`. All automated tests stub Strava entirely (in-memory fakes + `httpx.MockTransport`), so the build/test loop is fully local; the **real** OAuth click-through is verified only on the deployed Render stack (a local backend can't receive an onrender redirect).
- Frontend conventions are the contract in `frontend/CLAUDE.md` (folder layout, `@/` imports, data via `api/` hooks, token utilities for styling, co-located Vitest tests). Backend uses sync FastAPI endpoints and `def` handlers.
- Spec of record: `docs/superpowers/specs/2026-06-20-peakstats-design.md`.
- Frequent commits: each task ends in a commit. TDD where behavior exists.

```python
from itsdangerous import BadSignature, SignatureExpired, URLSafeTimedSerializer

SESSION_COOKIE = "ps_session"
STATE_COOKIE = "ps_oauth_state"
SESSION_MAX_AGE = 60 * 60 * 24 * 30  # 30 days, in seconds


def _serializer(secret: str) -> URLSafeTimedSerializer:
    return URLSafeTimedSerializer(secret, salt="ps-session")


def sign_session(athlete_id: int, secret: str) -> str:
    return _serializer(secret).dumps({"athlete_id": athlete_id})


def read_session(
    token: str, secret: str, max_age: int = SESSION_MAX_AGE
) -> int | None:
    try:
        data = _serializer(secret).loads(token, max_age=max_age)
    except (BadSignature, SignatureExpired):
        return None
    return int(data["athlete_id"])
```

Create `backend/app/strava.py`:
```python
from dataclasses import dataclass, field
from datetime import datetime, timezone
from urllib.parse import urlencode

import httpx

AUTHORIZE_URL = "https://www.strava.com/oauth/authorize"
TOKEN_URL = "https://www.strava.com/oauth/token"
DEAUTHORIZE_URL = "https://www.strava.com/oauth/deauthorize"
SCOPE = "read,activity:read_all"


@dataclass
class StravaToken:
    access_token: str
    refresh_token: str
    expires_at: datetime
    athlete: dict = field(default_factory=dict)


class StravaClient:
    def __init__(
        self,
        http: httpx.Client,
        client_id: str,
        client_secret: str,
        redirect_uri: str,
    ) -> None:
        self._http = http
        self._client_id = client_id
        self._client_secret = client_secret
        self._redirect_uri = redirect_uri

    def authorize_url(self, state: str) -> str:
        params = urlencode(
            {
                "client_id": self._client_id,
                "redirect_uri": self._redirect_uri,
                "response_type": "code",
                "approval_prompt": "auto",
                "scope": SCOPE,
                "state": state,
            }
        )
        return f"{AUTHORIZE_URL}?{params}"

    def _post_token(self, data: dict) -> StravaToken:
        response = self._http.post(TOKEN_URL, data=data)
        response.raise_for_status()
        payload = response.json()
        return StravaToken(
            access_token=payload["access_token"],
            refresh_token=payload["refresh_token"],
            expires_at=datetime.fromtimestamp(payload["expires_at"], tz=timezone.utc),
            athlete=payload.get("athlete", {}),
        )

    def exchange_code(self, code: str) -> StravaToken:
        return self._post_token(
            {
                "client_id": self._client_id,
                "client_secret": self._client_secret,
                "code": code,
                "grant_type": "authorization_code",
            }
        )

    def refresh(self, refresh_token: str) -> StravaToken:
        return self._post_token(
            {
                "client_id": self._client_id,
                "client_secret": self._client_secret,
                "refresh_token": refresh_token,
                "grant_type": "refresh_token",
            }
        )

    def deauthorize(self, access_token: str) -> None:
        response = self._http.post(DEAUTHORIZE_URL, data={"access_token": access_token})
        response.raise_for_status()
```

```python
from datetime import datetime

import httpx


class Db:
    def __init__(self, http: httpx.Client, base_url: str, service_key: str) -> None:
        self._http = http
        self._rest = f"{base_url}/rest/v1"
        self._headers = {
            "apikey": service_key,
            "Authorization": f"Bearer {service_key}",
            "Content-Type": "application/json",
        }

    def _upsert(self, table: str, conflict: str, row: dict) -> None:
        response = self._http.post(
            f"{self._rest}/{table}",
            params={"on_conflict": conflict},
            headers={**self._headers, "Prefer": "resolution=merge-duplicates"},
            json=[row],
        )
        response.raise_for_status()

    def _get_one(self, table: str, athlete_key: str, athlete_id: int) -> dict | None:
        response = self._http.get(
            f"{self._rest}/{table}",
            params={athlete_key: f"eq.{athlete_id}", "select": "*"},
            headers=self._headers,
        )
        response.raise_for_status()
        rows = response.json()
        return rows[0] if rows else None

    def upsert_athlete(self, athlete_id: int, name: str, avatar_url: str | None) -> None:
        self._upsert(
            "athletes",
            "id",
            {"id": athlete_id, "name": name, "avatar_url": avatar_url},
        )

    def upsert_tokens(
        self,
        athlete_id: int,
        access_token: str,
        refresh_token: str,
        expires_at: datetime,
    ) -> None:
        self._upsert(
            "strava_tokens",
            "athlete_id",
            {
                "athlete_id": athlete_id,
                "access_token": access_token,
                "refresh_token": refresh_token,
                "expires_at": expires_at.isoformat(),
            },
        )

    def get_athlete(self, athlete_id: int) -> dict | None:
        return self._get_one("athletes", "id", athlete_id)

    def get_tokens(self, athlete_id: int) -> dict | None:
        return self._get_one("strava_tokens", "athlete_id", athlete_id)

    def delete_athlete(self, athlete_id: int) -> None:
        response = self._http.request(
            "DELETE",
            f"{self._rest}/athletes",
            params={"id": f"eq.{athlete_id}"},
            headers=self._headers,
        )
        response.raise_for_status()
```

```python
from collections.abc import Iterator

import httpx
from fastapi import Depends

from app.config import Settings, get_settings
from app.db import Db
from app.strava import StravaClient


def get_strava_client(
    settings: Settings = Depends(get_settings),
) -> Iterator[StravaClient]:
    redirect_uri = f"{settings.backend_base_url}/auth/strava/callback"
    with httpx.Client(timeout=10) as http:
        yield StravaClient(
            http,
            settings.strava_client_id,
            settings.strava_client_secret,
            redirect_uri,
        )


def get_db(settings: Settings = Depends(get_settings)) -> Iterator[Db]:
    with httpx.Client(timeout=10) as http:
        yield Db(http, settings.supabase_url, settings.supabase_service_role_key)
```

```python
import secrets

from fastapi import APIRouter, Depends, Request, Response
from fastapi.responses import RedirectResponse

from app.config import Settings, get_settings
from app.deps import get_db, get_strava_client
from app.db import Db
from app.session import (
    SESSION_COOKIE,
    SESSION_MAX_AGE,
    STATE_COOKIE,
    sign_session,
)
from app.strava import StravaClient

router = APIRouter(tags=["auth"])


def _set_cookie(
    response: Response, name: str, value: str, settings: Settings, max_age: int
) -> None:
    response.set_cookie(
        key=name,
        value=value,
        max_age=max_age,
        httponly=True,
        secure=settings.session_cookie_secure,
        samesite=settings.session_cookie_samesite,
        path="/",
    )


@router.get("/auth/strava/login")
def login(
    settings: Settings = Depends(get_settings),
    strava: StravaClient = Depends(get_strava_client),
) -> Response:
    state = secrets.token_urlsafe(32)
    response = RedirectResponse(strava.authorize_url(state), status_code=302)
    _set_cookie(response, STATE_COOKIE, state, settings, max_age=600)
    return response


@router.get("/auth/strava/callback")
def callback(
    request: Request,
    code: str | None = None,
    state: str | None = None,
    error: str | None = None,
    settings: Settings = Depends(get_settings),
    strava: StravaClient = Depends(get_strava_client),
    db: Db = Depends(get_db),
) -> Response:
    cookie_state = request.cookies.get(STATE_COOKIE)
    state_ok = bool(state and cookie_state and secrets.compare_digest(state, cookie_state))
    if error or not code or not state_ok:
        return RedirectResponse(f"{settings.frontend_origin}/?auth=error", status_code=302)

    token = strava.exchange_code(code)
    athlete = token.athlete
    name = f"{athlete.get('firstname', '')} {athlete.get('lastname', '')}".strip()
    db.upsert_athlete(athlete["id"], name, athlete.get("profile"))
    db.upsert_tokens(
        athlete["id"], token.access_token, token.refresh_token, token.expires_at
    )

    response = RedirectResponse(f"{settings.frontend_origin}/app", status_code=302)
    _set_cookie(
        response,
        SESSION_COOKIE,
        sign_session(athlete["id"], settings.session_secret),
        settings,
        max_age=SESSION_MAX_AGE,
    )
    response.delete_cookie(STATE_COOKIE, path="/")
    return response
```

```ts
import { useEffect, useState } from "react";
import { config } from "@/lib/config";
import type { Athlete } from "@/types/athlete";
import { apiFetch } from "./client";

/** Full-page navigation target that starts Strava OAuth on the backend. */
export const stravaLoginUrl = `${config.apiBaseUrl}/auth/strava/login`;

export function fetchAthlete(): Promise<Athlete> {
  return apiFetch<Athlete>("/athlete");
}

export function logout(): Promise<void> {
  return apiFetch<void>("/auth/logout", { method: "POST" });
}

export function disconnect(): Promise<void> {
  return apiFetch<void>("/athlete/connection", { method: "DELETE" });
}

export type UseAthlete = {
  data: Athlete | null;
  isLoading: boolean;
  error: Error | null;
};

/** Loads the current athlete once on mount; a 401 surfaces as `error`. */
export function useAthlete(): UseAthlete {
  const [data, setData] = useState<Athlete | null>(null);
  const [error, setError] = useState<Error | null>(null);

  useEffect(() => {
    let active = true;
    fetchAthlete().then(
      (athlete) => active && setData(athlete),
      (err) => active && setError(err as Error)
    );
    return () => {
      active = false;
    };
  }, []);

  return { data, isLoading: data === null && error === null, error };
}
```

- [ ] **Step 4: Implement disconnect**

In `backend/app/routers/athlete.py`, replace the import block and append the route:
```python
from datetime import datetime, timedelta, timezone

import httpx
from fastapi import APIRouter, Depends, HTTPException, Response

from app.auth import get_current_athlete_id
from app.config import Settings, get_settings
from app.deps import get_db, get_strava_client
from app.db import Db
from app.session import SESSION_COOKIE
from app.strava import StravaClient
```
```python
@router.delete("/athlete/connection", status_code=204)
def disconnect(
    athlete_id: int = Depends(get_current_athlete_id),
    db: Db = Depends(get_db),
    strava: StravaClient = Depends(get_strava_client),
    settings: Settings = Depends(get_settings),
) -> Response:
    tokens = db.get_tokens(athlete_id)
    if tokens:
        access_token = tokens["access_token"]
        expires_at = datetime.fromisoformat(tokens["expires_at"])
        if expires_at <= datetime.now(timezone.utc) + timedelta(seconds=60):
            access_token = strava.refresh(tokens["refresh_token"]).access_token
        try:
            strava.deauthorize(access_token)
        except httpx.HTTPError:
            pass  # best-effort: still drop local data even if Strava is unreachable
    db.delete_athlete(athlete_id)

    response = Response(status_code=204)
    response.delete_cookie(
        SESSION_COOKIE,
        path="/",
        secure=settings.session_cookie_secure,
        samesite=settings.session_cookie_samesite,
    )
    return response
```
